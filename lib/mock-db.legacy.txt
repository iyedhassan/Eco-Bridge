function createSeedStore(): DataStore {
  const listings = seedListings()
  const projects = seedProjects
  const matches: Match[] = listings.map((listing) => ({
    id: id('match'),
    listingId: listing.id,
    projectId: projects[0].id,
    buyerId: projects[0].ownerId,
    score: computeMatchScore({ listing, project: projects[0] }),
    createdAt: now(),
  }))

  return {
    users: seedUsers,
    categories: baseCategories,
    materials: baseMaterials,
    listings,
    projects,
    matches,
    notifications: [
      {
        id: id('notif'),
        userId: 'user_wood_buyer',

export function dbGetCategories() {
  return store.categories
}

export function dbGetMaterials() {
  return store.materials
}

export function dbGetListings(filters: ListingFilters = {}) {
  return store.listings.filter((listing) => {
    if (filters.materialType && !listing.materialType.toLowerCase().includes(filters.materialType.toLowerCase())) return false
    if (filters.industry && listing.industry.toLowerCase() !== filters.industry.toLowerCase()) return false
    if (filters.location && !listing.location.toLowerCase().includes(filters.location.toLowerCase())) return false
    if (typeof filters.minPrice === 'number' && listing.pricePerUnit < filters.minPrice) return false
    if (typeof filters.maxPrice === 'number' && listing.pricePerUnit > filters.maxPrice) return false
    if (typeof filters.minQuantity === 'number' && listing.quantity < filters.minQuantity) return false
    if (filters.availability && listing.availability !== filters.availability) return false
    return true
  })
}

export function dbGetListingById(idValue: string) {
  return store.listings.find((listing) => listing.id === idValue)
}

export function dbCreateListing(input: Omit<Listing, 'id' | 'createdAt'>) {
  const listing: Listing = {
    id: id('list'),
    ...input,
    impactEstimate: input.impactEstimate ?? calculateListingImpact(input),
    createdAt: now(),
  }

  store.listings.unshift(listing)

  for (const savedSearch of store.savedSearches) {
    const matchesSearch = dbGetListings(savedSearch.filters).some((item) => item.id === listing.id)
    if (!matchesSearch) continue

    store.notifications.unshift({
      id: id('notif'),
      userId: savedSearch.userId,
      type: 'new-listing',
      title: 'إعلان جديد يطابق بحثك المحفوظ',
      message: `${listing.title} متوفر حالياً في جهة ${listing.location}.`,
      read: false,
      createdAt: now(),
    })
  }

  return listing
}

export function dbUpdateListing(idValue: string, input: Partial<Listing>) {
  const index = store.listings.findIndex((listing) => listing.id === idValue)
  if (index < 0) return null

  store.listings[index] = { ...store.listings[index], ...input }
  return store.listings[index]
}

export function dbDeleteListing(idValue: string) {
  const before = store.listings.length
  store.listings = store.listings.filter((listing) => listing.id !== idValue)
  return store.listings.length < before
}

export function dbCreateProject(input: Omit<Project, 'id' | 'createdAt'>) {
  const project: Project = {
    id: id('proj'),
    ...input,
    createdAt: now(),
  }

  store.projects.unshift(project)
  return project
}

export function dbGetProjectsByUser(userId: string) {
  return store.projects.filter((project) => project.ownerId === userId)
}

export function dbGetMatchesForUser(userId: string) {
  const ownProjects = store.projects.filter((project) => project.ownerId === userId).map((project) => project.id)
  return store.matches
    .filter((match) => ownProjects.includes(match.projectId) || match.buyerId === userId)
    .sort((a, b) => b.score.total - a.score.total)
}

export function dbGenerateMatchesForProject(project: Project) {
  const generated = store.listings.map((listing) => ({
    id: id('match'),
    listingId: listing.id,
    projectId: project.id,
    buyerId: project.ownerId,
    score: computeMatchScore({ listing, project }),
    createdAt: now(),
  }))

  store.matches.unshift(...generated)

  for (const match of generated.slice(0, 8)) {
    const listing = dbGetListingById(match.listingId)
    if (!listing) continue

    store.notifications.unshift(
      {
        id: id('notif'),
        userId: project.ownerId,
        type: 'new-match',
        title: 'تم العثور على فرصة مطابقة',
        message: `الإعلان "${listing.title}" تحصل على نسبة توافق ${match.score.total}/100 مع مشروعك.` ,
        read: false,
        createdAt: now(),
      },
      {
        id: id('notif'),
        userId: listing.sellerId,
        type: 'interest',
        title: 'مستثمر مهتم بفواضلك',
        message: `مشروع من قطاع ${project.industry} قد يحتاج إلى "${listing.title}".`,
        read: false,
        createdAt: now(),
      },
    )
  }

  return generated.sort((a, b) => b.score.total - a.score.total)
}

export function dbGetNotifications(userId: string) {
  return store.notifications.filter((notification) => notification.userId === userId)
}

export function dbMarkNotificationRead(idValue: string) {
  const item = store.notifications.find((notification) => notification.id === idValue)
  if (!item) return null
  item.read = true
  return item
}

export function dbCreateSavedSearch(input: Omit<SavedSearch, 'id' | 'createdAt'>) {
  const search: SavedSearch = {
    id: id('saved'),
    ...input,
    createdAt: now(),
  }

  store.savedSearches.unshift(search)
  return search
}

export function dbGetDashboardStats(userId: string): DashboardStats {
  const userMatches = dbGetMatchesForUser(userId)
  const userAlerts = dbGetNotifications(userId)
  const userSaved = store.savedSearches.filter((search) => search.userId === userId)

  const topListings = userMatches.slice(0, 5).map((match) => dbGetListingById(match.listingId)).filter(Boolean) as Listing[]

  const portfolioImpact = calculatePortfolioImpact(topListings)

  return {
    totalListings: store.listings.length,
    matchedOpportunities: userMatches.length,
    savedSearches: userSaved.length,
    alerts: userAlerts.filter((alert) => !alert.read).length,
    ecoImpact: {
      wasteReusedKg: portfolioImpact.wasteReusedKg,
      moneySavedTnd: Math.round(portfolioImpact.moneySavedTnd),
      co2ReductionKg: portfolioImpact.co2ReductionKg,
      storageAvoidedKg: portfolioImpact.storageAvoidedKg,
    },
  }
}
